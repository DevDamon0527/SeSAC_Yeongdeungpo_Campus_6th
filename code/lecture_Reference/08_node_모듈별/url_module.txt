var url = require('url');
var addr = 'http://localhost:8000/example.html?year=2024&month=february';
var q = url.parse(addr, true);

console.log(q.host); //returns 'localhost:8000'
console.log(q.pathname); //returns '/example.htm'
console.log(q.search); //returns '?year=2024&month=february'

var qdata = q.query; //returns an object: { year: 2024, month: 'february' }
console.log(qdata.month); //returns 'february'
